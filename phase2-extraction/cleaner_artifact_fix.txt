    def _remove_ocr_artifacts(self, text: str) -> str:
        """Remove OCR noise like single letters on their own lines."""
        # Remove single-letter lines (common OCR error)
        text = re.sub(r'^\s*[a-zA-Z]\s*$', '', text, flags=re.MULTILINE)
        
        # Fix spaced-out words (e.g., "T h e" â†’ "The")
        # Use a DIFFERENT strategy: Look for patterns where single letters
        # are separated by single spaces, and collapse ONLY those groups
        
        # Pattern explanation:
        # (\b[a-zA-Z]\b) - single letter word (with word boundaries)
        # (?:\s+\b[a-zA-Z]\b){2,} - followed by 2+ more single-letter words
        # This will match "T h e" but NOT "The Gift" because "The" and "Gift" 
        # are multi-letter words
        
        def collapse_single_letters(match):
            # Remove ALL spaces from the matched section
            return match.group(0).replace(' ', '')
        
        # Match: single letter, space, single letter, space, single letter (3+ total)
        # The \b ensures we only match complete single-letter "words"
        text = re.sub(r'\b[a-zA-Z]\b(?:\s+\b[a-zA-Z]\b){2,}', 
                      collapse_single_letters, 
                      text)
        
        return text
